22def extract_opencore_packages(opencore_pkg: Union[str, Path], 

23 target: str, 

24 resolvers: List[dict], 

25 packages: dict, 

26 ) -> dict: 

27 """Extracts build entries as vendored packages from an OpenCore package. 

28 

29 Args: 

30 opencore_pkg: Path to an existing OpenCore package. 

31 target: The desired target architecture of the OpenCore EFI. 

32 resolvers: The list of resolver entries to update. 

33 packages: The list of packages to update. 

34 

35 Returns: 

36 A dictionary of extracted build entries. 

37 """ 

38 

39 # Replace archive temporary directory with extracted contents 

40 opencore.extract_opencore_archive(opencore_pkg, target) 

41 # Extract additional OpenCore binaries not shipped in the main package 

42 if binary_pkg := nested_get(packages, ['OpenCorePkg', 'OcBinaryData']): 

43 opencore.extract_ocbinary_archive(pkg=binary_pkg, oc_pkg=opencore_pkg) 

44 nested_del(packages, ['OpenCorePkg', 'OcBinaryData']) 

45 

46 # Cleanup resolver entries 

47 prune_resolver_entry(resolvers, key='__category', value='OpenCorePkg') 

48 

49 # Extract build entries from the OpenCore package as (vendored) packages 

50 extracted = opencore.extract_build_entries(opencore_pkg, resolvers) 

51 

52 return extracted

55 """Iterate over the entries in the build configuration.""" 

56 def group_entries(category: str, entries: dict): 

57 return [(category, name, tmpdir) for name, tmpdir in entries.items()] 

58 return list(chain(*[group_entries(c,e) for c,e in unpacked_entries.items()]))

60def extract_build_packages(build_vars: dict, 

61 build_config: dict, 

62 resolvers: List[dict], 

63 packages: dict, 

64 build_dir: Path, 

65 *args, 

66 __wrapper: Optional[Iterator]=None, 

67 **kwargs 

68 ) -> dict: 

69 """Extracts build entries from unpacked packages. 

70 

71 Args: 

72 build_vars: The configured build variables. 

73 build_config: The configured build specification. 

74 resolvers: The list of resolver entries to update. 

75 packages: The list of packages to extract. 

76 build_dir: The path to the build directory. 

77 *args: Additional arguments to pass to the iterator. 

78 __wrapper: A wrapper function to apply to the iterator. (Optional) 

79 **kwargs: Additional keyword arguments to pass to the iterator. 

80 

81 Returns: 

82 A dictionary of extracted build entries. 

83 """ 

84 

85 def _get_resolver_entry(category: str, name: str) -> Union[dict, None]: 

86 return next(filter(lambda e: e['__category'] == category and 

87 e['name'] == name, 

88 resolvers), None) 

89 

90 # Extract build variables 

91 default_build = build_vars['variables']['build'] 

92 

93 # Handle interactive mode for iterator 

94 iterator = _iterate_extract_packages(packages) 

95 if __wrapper is not None: iterator = __wrapper(iterator, *args, **kwargs) 

96 

97 # Extract build entries from the remaining packages 

98 extracted_entries = {} 

99 for (category, name, tmpdir) in iterator: 

100 ext, _ = _category_extension(category) 

101 resolver_entry = _get_resolver_entry(category, name) 

102 # Extract SSDTs from the archive 

103 if category == 'ACPI': 

104 extract = ssdts.extract_ssdts(tmpdir) 

105 # Extract kexts from the archive 

106 elif category == 'Kexts': 

107 entry_cfg = nested_get(build_config, ['Kexts', resolver_entry['name']], {}) 

108 entry_build = resolver_entry.get('build') or default_build 

109 extract = kexts.extract_kexts(tmpdir, build=entry_build) 

110 # Filter out plugins that are not bundled 

111 for k_name, kext in extract.copy().items(): 

112 # Exclude plugins that are already bundled 

113 if is_plugin := '.kext/' in kext['__path']: 

114 nested_del(extract, [k_name]) 

115 # Prune implicitly excluded plugins 

116 if k_name not in entry_cfg.get('bundled', []): 

117 # Prune plugins that don't match a wildcard specifier 

118 plugin_entry = build_config['Kexts'].get(k_name, {}) 

119 if plugin_entry.get('specifier') != "*": 

120 remove(kext['__extracted']) 

121 else: continue 

122 # Extract drivers or tools from the archive 

123 elif category in ('Drivers', 'Tools'): 

124 extract = {} 

125 for binary_path in glob(tmpdir, f'**/*{ext}'): 

126 path = f'.{binary_path.as_posix().split(tmpdir.as_posix())[1]}' 

127 extract[binary_path.name] = { 

128 '__extracted': binary_path, 

129 '__path': path 

130 } 

131 # Extract resources from the archive 

132 elif category == 'Resources': 

133 pass 

134 

135 # Update extracted paths 

136 for k,e in extract.items(): 

137 e_name = name if len(extract) == 1 else k 

138 # Ensure only valid build entries are extracted 

139 if _get_resolver_entry(category, e_name): 

140 e['__dest'] = build_dir.joinpath('EFI', 'OC', category, f'{e_name}{ext}') 

141 nested_set(extracted_entries, [category, e_name], e) 

142 

143 return extracted_entries

145def _iterate_prune_packages(extracted_entries: dict): 

146 """Iterate over the entries in the build configuration.""" 

147 def group_entries(category: str, entries: dict): 

148 return [(category, name, entry) for name, entry in entries.items()] 

149 return list(chain(*[group_entries(c,e) for c,e in extracted_entries.items()]))

151def prune_build_packages(build_config: dict, 

152 extracted_entries: dict, 

153 *args, 

154 __wrapper: Optional[Iterator]=None, 

155 **kwargs 

156 ) -> dict: 

157 """Prunes the build configuration of entries that were not extracted. 

158 

159 Args: 

160 build_config: The build configuration. 

161 extracted_entries: The extracted build entries. 

162 *args: Additional arguments to pass to the iterator. 

163 __wrapper: A wrapper function to apply to the iterator. (Optional) 

164 **kwargs: Additional keyword arguments to pass to the iterator. 

165 

166 Returns: 

167 A dictionary of pruned build entries. 

168 """ 

169 

170 # Create list of entry names and their bundled package names 

171 entries = reduce(merge_dict, [ 

172 { c: (k, *nested_get(e, ['bundled'], default=[])) } 

173 for c,d in build_config.items() 

174 for k,e in d.items() 

175 ], {}) 

176 

177 # Handle interactive mode for iterator 

178 iterator = _iterate_prune_packages(extracted_entries) 

179 if __wrapper is not None: iterator = __wrapper(iterator, *args, **kwargs) 

180 

181 for category, name, entry in iterator: 

182 # Prune extracted entries that are not in the build configuration 

183 if not name in nested_get(entries, [category], default=()): 

184 nested_del(extracted_entries, [category, name]) 

185 remove(entry['__extracted'])
