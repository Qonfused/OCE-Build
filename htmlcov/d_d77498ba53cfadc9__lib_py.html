14TAGS = ('@append', '@delete', '@fallback', '@override', '@prepend') 

15"""Preprocessor tags for controlling output dict semantics."""

17def _append_tags(cursor: dict, 

18 frontmatter_dict: dict, 

19 defer_tree: bool=False 

20 ) -> None: 

21 """Append tag to frontmatter if entries marked with tag. 

22 

23 Args: 

24 cursor: The cursor dictionary. 

25 frontmatter_dict: The frontmatter dictionary. 

26 defer_tree: Whether to defer tree tag resolution until the next call. 

27 """ 

28 

29 if cursor['has_tag'] is None: return 

30 tag_name, tag_options = cursor['has_tag'] 

31 

32 # Leave tag tree null until another tag discovers the tree tag 

33 if defer_tree: 

34 if not 'defer_tree' in cursor: cursor['defer_tree'] = [] 

35 cursor['defer_tree'].append(len(frontmatter_dict['tags'])) 

36 tag_tree = None 

37 else: 

38 # Update deferred trees with current tree value 

39 if 'defer_tree' in cursor: 

40 for deferred in cursor['defer_tree']: 

41 frontmatter_dict['tags'][deferred][1] = deepcopy(cursor['tag_tree']) 

42 del cursor['defer_tree'] 

43 tag_tree = deepcopy(cursor['tag_tree']) 

44 

45 frontmatter_dict['tags'].append([tag_name, tag_tree, tag_options]) 

46 # Reset cursor attr 

47 cursor['has_tag'] = None

49def _apply_macro(macro: str, 

50 flags: List[str], 

51 cursor: dict, 

52 frontmatter_dict: dict 

53 ) -> None: 

54 """Applies preprocessor macros to parser cursor or frontmatter. 

55 

56 Args: 

57 macro: The macro to apply. 

58 flags: The current flags. 

59 tokens: The parsed tokens. 

60 cursor: The cursor dictionary. 

61 frontmatter_dict: The frontmatter dictionary. 

62 """ 

63 

64 # Check if macro has flag 

65 flag = re_search(r'\((.*)\)', macro, group=1) 

66 if flag is not None: macro = macro[:-len(f'({flag})')] 

67 

68 # Mark tagged entries on cursor (to append to frontmatter) 

69 if any(macro.startswith(t) for t in TAGS): 

70 # Handle any unresolved tags (non-attached) 

71 if cursor['has_tag'] is not None: 

72 _append_tags(cursor, frontmatter_dict, defer_tree=True) 

73 cursor['has_tag'] = (macro, flag) 

74 # Check if flag exists 

75 elif macro == '@ifdef': 

76 is_defined = (flag in flags) \ 

77 or (flag in frontmatter_dict) 

78 cursor['skip'] = not is_defined 

79 elif macro == '@ifndef': 

80 is_not_defined = (flag not in flags) \ 

81 and (flag not in frontmatter_dict) 

82 cursor['skip'] = not is_not_defined 

83 # Check if flag meets conditional 

84 # elif macro == '@if': 

85 # elif macro == '@elif': 

86 # Switch macro skip 

87 elif macro == '@else': 

88 cursor['skip'] = not cursor['skip'] 

89 # End macro checking scope 

90 elif macro == '@endif': 

91 cursor['skip'] = False

93def update_cursor(level: int, 

94 key: str, 

95 cursor: dict, 

96 upshift: int=0, 

97 ) -> None: 

98 """Updates the cursor dictionary. 

99 

100 Args: 

101 level: The level of the cursor. 

102 key: The key of the cursor. 

103 cursor: The cursor dictionary. 

104 uplevel: Number of levels to ignore 

105 """ 

106 bias = cursor['indent'] if upshift else 1 

107 # Handle key (cursor) resets 

108 if level < 2: cursor['keys'] = [key] 

109 # Handle key (cursor) movement 

110 else: 

111 if level == cursor['level'] and len(cursor['keys']): 

112 cursor['keys'].pop(-1) 

113 elif cursor['keys'] and level < cursor['level']: 

114 downlevel = (cursor['level'] - level) / cursor['indent'] 

115 cursor['keys'] = cursor['keys'][:-int(downlevel + upshift)] 

116 cursor['keys'].append(key) 

117 # Update cursor 

118 cursor['level'] = level 

119 cursor['indent'] = max(bias, level / max(1,(len(cursor['keys']) - upshift)))
